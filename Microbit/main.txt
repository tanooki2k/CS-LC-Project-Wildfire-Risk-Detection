def Blink(repeat: number, delay: number):
    for index in range(repeat):
        Risk_Level_LEDs(0, 0, 0, 0)
        basic.pause(delay)

def SoundAlarm(pitch: number, delay2: number):
    music.stop_all_sounds()
    basic.pause(10)
    music.play(music.tone_playable(pitch, music.beat(BeatFraction.WHOLE)),
        music.PlaybackMode.IN_BACKGROUND)
    basic.pause(delay2)
    music.play(music.tone_playable(pitch, music.beat(BeatFraction.WHOLE)),
        music.PlaybackMode.UNTIL_DONE)

def on_log_full():
    basic.show_icon(IconNames.YES)
    if input.button_is_pressed(Button.AB):
        datalogger.delete_log()
datalogger.on_log_full(on_log_full)

def on_button_pressed_a():
    global global_time, is_started
    global_time = input.running_time()
    is_started = True
    OnOff_LED(1023)
input.on_button_pressed(Button.A, on_button_pressed_a)

def Risk_Level_LEDs(green: number, yellow: number, orange: number, red: number):
    pins.analog_write_pin(AnalogPin.P3, green)
    pins.analog_write_pin(AnalogPin.P4, yellow)
    pins.analog_write_pin(AnalogPin.P5, orange)
    pins.analog_write_pin(AnalogPin.P6, red)

def digitliseMoisture(moistValue: number):
    if moistValue < wetnessConstant:
        return 0
    else:
        return 1
def OnOff_LED(on: number):
    pins.analog_write_pin(AnalogPin.P7, on)

def on_button_pressed_b():
    global elapsed_time, is_started
    if is_started:
        elapsed_time = input.running_time()
        is_started = False
        OnOff_LED(0)
        Risk_Level_LEDs(0, 0, 0, 0)
input.on_button_pressed(Button.B, on_button_pressed_b)

def on_data_received():
    global state, risk
    state = serial.read_line()
    if state == "low":
        risk = 0
    elif state == "moderate":
        risk = 1
    elif state == "high":
        risk = 2
    else:
        risk = 3
serial.on_data_received(serial.delimiters(Delimiters.NEW_LINE), on_data_received)

def on_forever():
    if is_started:
        if risk == 0:
            Risk_Level_LEDs(1023, 0, 0, 0)
        elif risk == 1:
            Risk_Level_LEDs(1023, 1023, 0, 0)
        elif risk == 2:
            Risk_Level_LEDs(1023, 1023, 1023, 0)
        else:
            Risk_Level_LEDs(1023, 1023, 1023, 1023)

def on_forever4():
    global prev_sate
    if state != prev_sate:
        prev_sate = state
        Blink(3, 250)
        if state == "low":
            SoundAlarm(262, 200)
        if state == "moderate":
            SoundAlarm(392, 150)
